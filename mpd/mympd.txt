### Ubuntu PPA

https://help.ubuntu.com/community/GnuPrivacyGuardHowto
https://help.launchpad.net/Packaging/PPA/BuildingASourcePackage

$ cd /home/parkmino/Downloads/ppa/mympd/
$ curl -s https://github.com/jcorporation/myMPD/releases/latest | cut -d\" -f2 | sed 's/releases\/tag/archive/; s/$/.tar.gz/' | wget --no-check-certificate "$(cat -)" -O ./mympd.tar.gz
$ export ver=$(curl -s https://github.com/jcorporation/myMPD/releases/latest | cut -d\" -f2 | sed 's/.*\/v//')
$ tar -xvzf mympd.tar.gz
$ mv mympd.tar.gz mympd_$ver.orig.tar.gz
$ mv myMPD-$ver/contrib/packaging/debian myMPD-$ver
$ cd myMPD-$ver

$ sed -i '1s/-1)/-1ppa1~focal)/; 1s/unstable/focal/' debian/changelog
$ cat debian/changelog
	mympd ($ver-1ppa1~focal) focal; urgency=medium
$ debuild -S -kF55443ED433D8977 # -sd
$ dput ppa:parkmino/mympd ../mympd_$ver-1ppa1~focal_source.changes

$ sed -i '1s/focal/bionic/g' debian/changelog
$ cat debian/changelog
	mympd ($ver-1ppa1~bionic) bionic; urgency=medium
$ debuild -S -kF55443ED433D8977 -sd
$ dput ppa:parkmino/mympd ../mympd_$ver-1ppa1~bionic_source.changes
