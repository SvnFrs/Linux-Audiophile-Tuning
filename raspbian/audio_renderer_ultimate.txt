Raspbian MPD Renderer

1. Minimal ALSA configuration

	$ sudo nano /usr/share/alsa/alsa.conf
		#pcm.default pcm.sound
		#ctl.control.type hw
		#ctl.control.card 1

2. Enable UPnP

	$ sudo nano /etc/mpd.conf
		#music_directory
		#playlist_directory
		#db_file
		#log_file
		#state_file
		#sticker_file
		database {
			plugin "upnp"
		}

3. Disable Upmpdcli

	$ sudo nano /etc/rc.local
		#... upmpdcli

4. Static IP

	$ sudo nano /etc/network/interfaces
		auto  eth0
		#iface eth0 inet manual
		iface eth0 inet static
			address		192.168.0.3
			netmask		255.255.255.0
			broadcast	192.168.0.255
			network		192.168.0.0
			gateway		192.168.0.1 
			dns-nameservers 168.126.63.1
	$ sudo systemctl disable dhcpcd.service

5. Remove unnecessay modules

	$ lsmod
	$ sudo nano /etc/modprobe.d/raspi-blacklist.conf
		 blacklist bcm2835_gpiomem
		 blacklist bcm2835_rng
		#blacklist ipv6
		 blacklist uio
		 blacklist uio-pdrv-genirq
		 blacklist i2c-dev
		#blacklist joydev
		#blacklist evdev

6. Remove unnecessay processes

	$ ps ax
	$ sudo systemctl disable cron ntp rsyslog avahi-daemon
	$ sudo nano /etc/rc.local
		for i in systemd-journald systemd-udevd systemd-logind triggerhappy getty@tty1 dbus sshd; do
		 service $i stop
		done

7. USB Tweak

	$ cp hub-ctrl /home/pi/
	$ sudo nano /etc/rc.local
		### Turn off USBs
		/home/pi/hub-ctrl -h 0 -P 5 -p 0
		/home/pi/hub-ctrl -h 0 -P 4 -p 0
		/home/pi/hub-ctrl -h 0 -P 2 -p 0 # Power off
