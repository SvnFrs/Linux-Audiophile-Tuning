### Build from GitHub ###

wget https://github.com/CDrummond/cantata/archive/master.zip
unzip -u master.zip -d ~/Downloads/mpd
mv master.zip ~/Downloads/mpd/cantata-master

if [ ! -d "~/Downloads/mpd/cantata-master/build" ]; then mkdir ~/Downloads/mpd/cantata-master/build ; fi ; cd ~/Downloads/mpd/cantata-master/build

or

cd ~/Downloads/mpd
rm -rf cantata
git clone git://github.com/CDrummond/cantata.git
cd cantata
mkdir build && cd build

cmake .. -DCMAKE_INSTALL_PREFIX=/usr -DCMAKE_BUILD_TYPE=Release -DENABLE_KDE=OFF -DENABLE_FFMPEG=OFF -DENABLE_MPG123=OFF -DENABLE_MTP=OFF -DENABLE_REMOTE_DEVICES=OFF -DENABLE_UDISKS2=OFF -DENABLE_QT5=OFF -DENABLE_EXTERNAL_TAGS=ON -DINSTALL_UBUNTU_ICONS=ON -DENABLE_HTTP_STREAM_PLAYBACK=OFF

sudo checkinstall -y --install=no --fstrans=no --pkgname=cantata-qt --pkgversion "1.9.50" --pkgrelease=$(date +"%Y.%m.%d") --requires=libc6,libcddb2,libcdparanoia0,libgcc1,libmusicbrainz5-0,libqjson0,libqt4-dbus,libqt4-network,libqtcore4,libqtgui4,libspeexdsp1,libstdc++6,libtag-extras1,libtag1c2a,libx11-6,phonon,zlib1g --backup=no --maintainer=parkmino@gmail.com

cp ~/Downloads/mpd/cantata-master/build/cantata*.deb ~/git/cantata

cd

### QT5 ###

sudo apt-get install qt5-default qttools5-dev-tools libgt5svg5-dev

cmake .. -DCMAKE_INSTALL_PREFIX=/usr -DCMAKE_BUILD_TYPE=Release -DENABLE_KDE=OFF -DENABLE_FFMPEG=OFF -DENABLE_MPG123=OFF -DENABLE_MTP=OFF -DENABLE_REMOTE_DEVICES=OFF -DENABLE_UDISKS2=OFF -DENABLE_QT5=ON -DENABLE_EXTERNAL_TAGS=ON -DINSTALL_UBUNTU_ICONS=ON -DENABLE_HTTP_STREAM_PLAYBACK=OFF

sudo checkinstall -y --install=no --fstrans=no --pkgname=cantata-qt5 --pkgversion "1.5.50" --pkgrelease=$(date +"%Y.%m.%d") --requires=libc6,libcddb2,libcdparanoia0,libgcc1,libmusicbrainz5-0,libqjson0,libqt5dbus5,libqt5network5,libqt5core5a,libqt5gui5,libspeexdsp1,libstdc++6,libtag-extras1,libtag1c2a,libx11-6,zlib1g --backup=no --maintainer=parkmino@gmail.com
