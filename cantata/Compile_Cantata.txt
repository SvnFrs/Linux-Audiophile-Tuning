### 1. Git Clone ###

rm -rf ~/Downloads/mpd/cantata-git

git clone git://github.com/CDrummond/cantata.git ~/Downloads/mpd/cantata-git

mkdir ~/Downloads/mpd/cantata-git/build && cd ~/Downloads/mpd/cantata-git/build

### 2. Svn Checkout ###

svn checkout https://github.com/CDrummond/cantata/trunk ~/Downloads/mpd/cantata-svn

svn update ~/Downloads/mpd/cantata-svn

if [ -d ~/Downloads/mpd/cantata-svn/build ] ; then rm -f ~/Downloads/mpd/cantata-svn/build/cantata-qt*.deb ; else mkdir ~/Downloads/mpd/cantata-svn/build ; fi ; cd ~/Downloads/mpd/cantata-svn/build

### 3. Wget ###

wget https://github.com/CDrummond/cantata/archive/master.zip

rm -rf ~/Downloads/mpd/cantata-master && unzip master.zip -d ~/Downloads/mpd && mv master.zip ~/Downloads/mpd/cantata-master && mkdir ~/Downloads/mpd/cantata-master/build && cd ~/Downloads/mpd/cantata-master/build

### Build ###

cmake .. -DCMAKE_INSTALL_PREFIX=/usr -DCMAKE_BUILD_TYPE=Release -DENABLE_FFMPEG=OFF -DENABLE_MPG123=OFF

sudo checkinstall -y --install=no --fstrans=no --pkgname=cantata-qt --pkgversion "2.0.50" --pkgrelease=$(date +"%Y.%m.%d") --backup=no --maintainer=parkmino@gmail.com

### Git Add ###

cd ~/git/audiophile
[ -f cantata/cantata-qt*amd64.deb ] && git rm cantata/cantata-qt*amd64.deb
cp $OLDPWD/cantata-qt*$(date +"%Y.%m.%d")*.deb cantata/
git add cantata/cantata-qt*$(date +"%Y.%m.%d")*.deb
git commit -m "Update Cantata package"
git push origin master

### Korean Translation Inquiry ###

svn log ~/Downloads/mpd/cantata-svn/po/cantata.pot | head -20
svn log ~/Downloads/mpd/cantata-svn/po/ko.po       | head -20
CANTATA_LANG=ko cantata
sudo cp /usr/share/cantata/translations/cantata_ko.qm /usr/share/cantata/translations/cantata_en.qm
diff ~/Documents/ko.po ~/Downloads/cantata-read-only/po/ko.po
